# 第17章 Function Calling 深入

> Function Calling 是 AI 与外部世界交互的核心机制。本章深入原理、实战和最佳实践。

## 17.1 什么是 Function Calling？

Function Calling 让 AI 模型能够**调用你定义的函数**，而不是只返回文本。

```
用户: "北京今天天气怎么样？"

没有 Function Calling:
  AI: "抱歉，我无法获取实时天气信息。"  ← 只能瞎说或拒绝

有 Function Calling:
  AI → 调用 get_weather(city="北京") → 返回数据 → AI 组织回复
  AI: "北京今天晴，25°C，适合出行。"  ← 基于真实数据回答
```

### 与传统 API 的区别

```
传统 API:  你写代码 → 调用 API → 处理结果
FC:        你定义函数 → AI 决定调用哪个 → AI 处理结果 → 回复用户
```

**AI 负责「决策」，你负责「执行」。**

## 17.2 OpenAI Function Calling

### 定义函数

```python
# code/ch17/openai_fc.py
from openai import OpenAI
import json

client = OpenAI()

# 定义工具
tools = [
    {
        "type": "function",
        "function": {
            "name": "get_weather",
            "description": "获取指定城市的当前天气",
            "parameters": {
                "type": "object",
                "properties": {
                    "city": {
                        "type": "string",
                        "description": "城市名称，如 '北京'、'上海'"
                    },
                    "unit": {
                        "type": "string",
                        "enum": ["celsius", "fahrenheit"],
                        "description": "温度单位，默认摄氏度"
                    }
                },
                "required": ["city"]
            }
        }
    },
    {
        "type": "function",
        "function": {
            "name": "search_products",
            "description": "搜索商品",
            "parameters": {
                "type": "object",
                "properties": {
                    "query": {
                        "type": "string",
                        "description": "搜索关键词"
                    },
                    "max_price": {
                        "type": "number",
                        "description": "最高价格"
                    },
                    "category": {
                        "type": "string",
                        "description": "商品类别"
                    }
                },
                "required": ["query"]
            }
        }
    }
]
```

### 调用流程

```python
# 函数实现
def get_weather(city: str, unit: str = "celsius") -> dict:
    """模拟天气查询"""
    data = {"北京": {"temp": 25, "condition": "晴"}}
    weather = data.get(city, {"temp": 20, "condition": "未知"})
    return {"city": city, "unit": unit, **weather}


def search_products(query: str, max_price: float = None, category: str = None) -> list:
    """模拟商品搜索"""
    return [{"name": f"{query}相关商品", "price": 99.9}]


# 完整调用流程
def chat_with_tools(user_message: str) -> str:
    messages = [{"role": "user", "content": user_message}]

    # 第一步：发送消息 + 工具定义
    response = client.chat.completions.create(
        model="gpt-4",
        messages=messages,
        tools=tools,
        tool_choice="auto"  # auto | none | required
    )

    msg = response.choices[0].message

    # 第二步：检查是否有工具调用
    if msg.tool_calls:
        # 第三步：执行工具
        messages.append(msg)  # 添加 assistant 消息

        for tool_call in msg.tool_calls:
            func_name = tool_call.function.name
            func_args = json.loads(tool_call.function.arguments)

            # 调用对应函数
            if func_name == "get_weather":
                result = get_weather(**func_args)
            elif func_name == "search_products":
                result = search_products(**func_args)
            else:
                result = {"error": f"未知函数: {func_name}"}

            # 第四步：把结果返回给模型
            messages.append({
                "role": "tool",
                "tool_call_id": tool_call.id,
                "content": json.dumps(result, ensure_ascii=False)
            })

        # 第五步：模型基于工具结果生成最终回复
        final_response = client.chat.completions.create(
            model="gpt-4",
            messages=messages
        )
        return final_response.choices[0].message.content

    # 没有工具调用，直接返回
    return msg.content


# 测试
print(chat_with_tools("北京今天天气怎么样？"))
# → "北京今天晴，25°C，适合出行。"
```

### 工具调用流程图

```
用户消息 + tools
       │
       ▼
┌──────────────┐
│  模型推理     │
│  需要工具？   │
└──────┬───────┘
       │
   ┌───┴───┐
   │       │
   否      是
   │       │
   ▼       ▼
返回文本  输出 tool_calls
              │
              ▼
       ┌──────────────┐
       │ 执行函数      │
       │ 获取结果      │
       └──────┬───────┘
              │
              ▼
       ┌──────────────┐
       │ 结果发回模型  │
       │ 生成最终回复  │
       └──────────────┘
```

## 17.3 Claude Tool Use

Claude 的工具调用语法更清晰：

```python
# code/ch17/claude_fc.py
import anthropic
import json

client = anthropic.Anthropic()

# 定义工具
tools = [
    {
        "name": "get_weather",
        "description": "获取指定城市的当前天气信息",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {
                    "type": "string",
                    "description": "城市名称"
                }
            },
            "required": ["city"]
        }
    }
]

# 函数实现
def get_weather(city: str) -> str:
    data = {"北京": "晴，25°C", "上海": "多云，22°C"}
    return data.get(city, f"{city}：暂无数据")


def chat_with_tools(user_message: str) -> str:
    messages = [{"role": "user", "content": user_message}]

    # 第一步：发送请求
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        tools=tools,
        messages=messages,
    )

    # 第二步：处理响应
    # Claude 的响应可能包含多种类型的 block
    tool_results = []

    for block in response.content:
        if block.type == "tool_use":
            # 执行工具
            result = get_weather(**block.input)
            tool_results.append({
                "type": "tool_result",
                "tool_use_id": block.id,
                "content": result,
            })

    # 如果有工具调用，把结果发回
    if tool_results:
        messages.append({"role": "assistant", "content": response.content})
        messages.append({"role": "user", "content": tool_results})

        # 第三步：最终回复
        final = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            tools=tools,
            messages=messages,
        )
        return final.content[0].text

    # 没有工具调用
    return response.content[0].text


print(chat_with_tools("北京今天天气怎么样？"))
```

### Claude vs OpenAI 对比

| 特性 | OpenAI | Claude |
|------|--------|--------|
| 工具定义位置 | `tools` 参数 | `tools` 参数 |
| 参数格式 | JSON Schema | JSON Schema |
| 结果返回 | `role: "tool"` + `tool_call_id` | `role: "user"` + `tool_result` |
| 多工具并行 | ✅ 支持 | ✅ 支持 |
| 强制调用 | `tool_choice: "required"` | `tool_choice: {"type": "any"}` |
| 禁止调用 | `tool_choice: "none"` | 不传 tools |

## 17.4 并行工具调用

一次请求返回多个工具调用：

```python
# OpenAI 并行调用
tools = [
    {"type": "function", "function": {"name": "get_weather", ...}},
    {"type": "function", "function": {"name": "get_time", ...}},
]

response = client.chat.completions.create(
    model="gpt-4",
    messages=[{"role": "user", "content": "北京天气怎么样？现在几点了？"}],
    tools=tools,
)

# 可能返回两个 tool_calls
msg = response.choices[0].message
if msg.tool_calls:
    for tc in msg.tool_calls:
        print(f"调用: {tc.function.name}({tc.function.arguments})")
    # 调用: get_weather({"city": "北京"})
    # 调用: get_time({})
```

## 17.5 流式 Function Calling

```python
# code/ch17/streaming_fc.py
from openai import OpenAI
import json

client = OpenAI()


def stream_with_tools(user_message: str, tools: list, functions: dict) -> str:
    """流式 Function Calling"""
    messages = [{"role": "user", "content": user_message}]

    stream = client.chat.completions.create(
        model="gpt-4",
        messages=messages,
        tools=tools,
        stream=True,
    )

    # 收集工具调用
    tool_calls = []
    current_call = None
    content_parts = []

    for chunk in stream:
        delta = chunk.choices[0].delta

        # 收集文本内容
        if delta.content:
            content_parts.append(delta.content)
            print(delta.content, end="", flush=True)

        # 收集工具调用
        if delta.tool_calls:
            for tc_delta in delta.tool_calls:
                if tc_delta.index is not None:
                    while len(tool_calls) <= tc_delta.index:
                        tool_calls.append({"id": "", "name": "", "arguments": ""})
                    current_call = tool_calls[tc_delta.index]

                if tc_delta.id:
                    current_call["id"] = tc_delta.id
                if tc_delta.function:
                    if tc_delta.function.name:
                        current_call["name"] = tc_delta.function.name
                    if tc_delta.function.arguments:
                        current_call["arguments"] += tc_delta.function.arguments

    # 执行工具调用
    if tool_calls:
        print(f"\n\n[调用 {len(tool_calls)} 个工具]")
        messages.append({"role": "assistant", "content": "".join(content_parts)})

        for tc in tool_calls:
            func = functions.get(tc["name"])
            if func:
                args = json.loads(tc["arguments"])
                result = func(**args)
                print(f"  {tc['name']}({args}) → {result}")

                messages.append({
                    "role": "tool",
                    "tool_call_id": tc["id"],
                    "content": json.dumps(result, ensure_ascii=False),
                })

        # 最终回复
        final = client.chat.completions.create(
            model="gpt-4", messages=messages
        )
        return final.choices[0].message.content

    return "".join(content_parts)
```

## 17.6 结构化输出

Function Calling 的副产品：强制模型输出结构化数据。

```python
# code/ch17/structured.py
import json
from openai import OpenAI

client = OpenAI()


def extract_info(text: str) -> dict:
    """从文本中提取结构化信息"""
    tools = [{
        "type": "function",
        "function": {
            "name": "save_extracted_info",
            "description": "保存从文本中提取的结构化信息",
            "parameters": {
                "type": "object",
                "properties": {
                    "people": {
                        "type": "array",
                        "items": {
                            "type": "object",
                            "properties": {
                                "name": {"type": "string"},
                                "role": {"type": "string"},
                                "company": {"type": "string"},
                            }
                        }
                    },
                    "dates": {
                        "type": "array",
                        "items": {"type": "string"}
                    },
                    "locations": {
                        "type": "array",
                        "items": {"type": "string"}
                    },
                    "summary": {"type": "string"}
                },
                "required": ["summary"]
            }
        }
    }]

    response = client.chat.completions.create(
        model="gpt-4",
        messages=[
            {"role": "system", "content": "从文本中提取所有结构化信息。"},
            {"role": "user", "content": text}
        ],
        tools=tools,
        tool_choice={"type": "function", "function": {"name": "save_extracted_info"}}
    )

    # 提取参数就是结构化结果
    tool_call = response.choices[0].message.tool_calls[0]
    return json.loads(tool_call.function.arguments)


# 测试
text = """
2024年3月15日，张三（ABC公司CTO）和李四在北京开会，
讨论了新产品发布计划，预计4月1日上线。
"""

result = extract_info(text)
print(json.dumps(result, ensure_ascii=False, indent=2))
# {
#   "people": [
#     {"name": "张三", "role": "CTO", "company": "ABC公司"},
#     {"name": "李四", "role": null, "company": null}
#   ],
#   "dates": ["2024-03-15", "2024-04-01"],
#   "locations": ["北京"],
#   "summary": "张三和李四在北京讨论新产品发布计划"
# }
```

## 17.7 工具定义最佳实践

### 1. 描述要充分

```python
# ❌ 差
{"name": "query", "description": "查询数据"}

# ✅ 好
{
    "name": "query_database",
    "description": "执行 SQL 查询并返回结果。仅支持 SELECT 语句。返回 JSON 格式的行数据。每次最多返回 100 行。",
}
```

### 2. 参数类型要精确

```python
# ❌ 差：类型模糊
{"price": {"type": "string", "description": "价格"}}

# ✅ 好：类型明确 + 约束
{
    "price": {
        "type": "number",
        "description": "价格（元），必须大于0",
        "minimum": 0
    }
}
```

### 3. 用 enum 限制选项

```python
{
    "sort_by": {
        "type": "string",
        "enum": ["price", "rating", "date"],
        "description": "排序字段"
    }
}
```

### 4. 函数要小而专

```python
# ❌ 一个函数做太多事
def process_data(action, data, format, ...):
    """处理数据"""

# ✅ 每个函数职责单一
def read_csv(path): ...
def filter_rows(data, condition): ...
def calculate_stats(data, column): ...
```

## 17.8 错误处理

```python
def safe_tool_call(func):
    """工具调用装饰器：捕获异常并返回错误信息"""
    def wrapper(*args, **kwargs):
        try:
            result = func(*args, **kwargs)
            return {"success": True, "data": result}
        except ValueError as e:
            return {"success": False, "error": f"参数错误: {e}"}
        except FileNotFoundError as e:
            return {"success": False, "error": f"文件不存在: {e}"}
        except Exception as e:
            return {"success": False, "error": f"执行失败: {type(e).__name__}: {e}"}
    return wrapper


# 注册工具时使用
@safe_tool_call
def read_file(path: str) -> str:
    return Path(path).read_text()


# 把错误信息返回给模型，让它自己处理
result = read_file("/not/exist")
# {"success": false, "error": "文件不存在: ..."}
# 模型会基于这个错误信息回复用户
```

## 17.9 Function Calling vs 其他方案

| 方案 | 原理 | 适用场景 |
|------|------|----------|
| **Function Calling** | 模型输出结构化调用 | 精确工具调用 |
| **JSON Mode** | 强制输出 JSON | 结构化数据提取 |
| **ReAct** | 思考+行动循环 | 复杂推理任务 |
| **Prompt Hacking** | 提示词中描述工具格式 | 简单场景、兼容性 |

## 本章小结

- Function Calling = AI 决策 + 你执行
- OpenAI 和 Claude 都支持，语法略有不同
- 支持并行调用和流式输出
- 结构化输出是 FC 的重要副产品
- 工具定义要清晰、精确、小而专
- 错误处理：返回结构化错误让模型自行处理
