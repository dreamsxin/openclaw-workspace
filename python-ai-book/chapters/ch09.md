# 第9章 记忆系统

> 让 AI 拥有长期记忆——记住你的偏好、历史对话、学到的知识。

## 9.1 记忆的三个层次

| 层次 | 存储位置 | 持久性 | 用途 |
|------|----------|--------|------|
| **工作记忆** | messages 列表 | 会话内 | 当前对话上下文 |
| **短期记忆** | 文件/数据库 | 天/周级 | 近期对话摘要 |
| **长期记忆** | 向量数据库 | 永久 | 知识库、用户画像 |

## 9.2 向量存储基础

### 什么是嵌入（Embedding）？

把文本转换为数字向量，语义相似的文本距离更近：

```python
# "猫" → [0.2, 0.8, 0.1, ...]
# "狗" → [0.3, 0.7, 0.2, ...]  # 与猫相近
# "汽车" → [0.9, 0.1, 0.8, ...]  # 与猫狗很远
```

### 使用嵌入模型

```python
# code/ch09/embeddings.py
from openai import OpenAI

client = OpenAI()


def get_embedding(text: str, model: str = "text-embedding-3-small") -> list[float]:
    """获取文本的嵌入向量"""
    response = client.embeddings.create(
        model=model,
        input=text
    )
    return response.data[0].embedding


# 比较相似度
import numpy as np


def cosine_similarity(a: list[float], b: list[float]) -> float:
    """计算余弦相似度"""
    a, b = np.array(a), np.array(b)
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))


# 示例
texts = ["Python编程", "JavaScript开发", "烹饪食谱"]
embeddings = [get_embedding(t) for t in texts]

print(f"Python vs JavaScript: {cosine_similarity(embeddings[0], embeddings[1]):.3f}")
print(f"Python vs 烹饪: {cosine_similarity(embeddings[0], embeddings[2]):.3f}")
```

## 9.3 ChromaDB：本地向量数据库

```python
# code/ch09/vector_store.py
import chromadb
from openai import OpenAI


class VectorMemory:
    """基于向量的长期记忆"""

    def __init__(self, collection_name: str = "memory"):
        self.client = chromadb.Client()
        self.collection = self.client.get_or_create_collection(
            name=collection_name,
            metadata={"hnsw:space": "cosine"}
        )
        self.openai = OpenAI()

    def remember(self, text: str, metadata: dict = None):
        """存储一条记忆"""
        embedding = self._embed(text)
        doc_id = f"doc_{self.collection.count() + 1}"

        self.collection.add(
            ids=[doc_id],
            embeddings=[embedding],
            documents=[text],
            metadatas=[metadata or {}]
        )

    def recall(self, query: str, n_results: int = 3) -> list[str]:
        """回忆与查询相关的记忆"""
        embedding = self._embed(query)
        results = self.collection.query(
            query_embeddings=[embedding],
            n_results=n_results
        )
        return results["documents"][0] if results["documents"] else []

    def forget(self, doc_id: str):
        """删除一条记忆"""
        self.collection.delete(ids=[doc_id])

    def _embed(self, text: str) -> list[float]:
        response = self.openai.embeddings.create(
            model="text-embedding-3-small",
            input=text
        )
        return response.data[0].embedding


# 使用
memory = VectorMemory()

# 存储记忆
memory.remember("用户喜欢用Python", {"category": "preference"})
memory.remember("用户在北京工作", {"category": "location"})
memory.remember("用户正在学习机器学习", {"category": "interest"})

# 回忆
results = memory.recall("用户喜欢什么编程语言？")
print(results)  # ['用户喜欢用Python']
```

## 9.4 RAG：检索增强生成

让 AI 基于你的知识库回答问题：

```python
# code/ch09/rag_system.py
from openai import OpenAI
from vector_store import VectorMemory


class RAGSystem:
    """RAG：检索增强生成"""

    def __init__(self):
        self.client = OpenAI()
        self.memory = VectorMemory("knowledge_base")

    def ingest(self, text: str, chunk_size: int = 500):
        """导入文档到知识库"""
        # 分块
        chunks = [text[i:i+chunk_size] for i in range(0, len(text), chunk_size)]
        for chunk in chunks:
            self.memory.remember(chunk)
        print(f"已导入 {len(chunks)} 个文本块")

    def query(self, question: str) -> str:
        """基于知识库回答问题"""
        # 1. 检索相关文档
        relevant_docs = self.memory.recall(question, n_results=3)
        context = "\n---\n".join(relevant_docs)

        # 2. 用检索结果增强 prompt
        prompt = f"""基于以下参考资料回答问题。如果资料中没有相关信息，请说明。

参考资料：
{context}

问题：{question}

要求：
- 只基于参考资料回答
- 引用具体来源
- 如果不确定，明确说明"""

        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "你是一个基于知识库的问答助手。"},
                {"role": "user", "content": prompt}
            ]
        )

        return response.choices[0].message.content


# 使用
rag = RAGSystem()

# 导入知识
rag.igest("""
Python是一种高级编程语言，由Guido van Rossum于1991年创建。
Python的设计哲学强调代码的可读性和简洁性。
Python支持多种编程范式，包括面向对象、函数式和过程式编程。
""")

# 提问
answer = rag.query("Python是谁创建的？")
print(answer)
```

## 9.5 混合记忆架构

结合多种记忆方式：

```python
# code/ch09/hybrid_memory.py
class HybridMemory:
    """混合记忆：对话 + 摘要 + 向量"""

    def __init__(self):
        self.working = []           # 工作记忆（当前对话）
        self.summaries = []         # 短期记忆（摘要）
        self.long_term = VectorMemory("long_term")  # 长期记忆

    def add_message(self, role: str, content: str):
        """添加消息"""
        self.working.append({"role": role, "content": content})

        # 超过10轮时自动摘要
        if len(self.working) > 20:
            self._compress()

    def recall(self, query: str) -> str:
        """综合回忆"""
        results = []

        # 1. 工作记忆（最近对话）
        recent = self.working[-6:]  # 最近3轮
        results.append("最近对话：\n" + self._format_messages(recent))

        # 2. 短期记忆（摘要）
        if self.summaries:
            results.append("对话摘要：\n" + "\n".join(self.summaries[-3:]))

        # 3. 长期记忆（向量检索）
        long_term = self.long_term.recall(query)
        if long_term:
            results.append("相关记忆：\n" + "\n".join(long_term))

        return "\n\n".join(results)

    def _compress(self):
        """压缩工作记忆到摘要"""
        # 用 AI 生成摘要
        conversation = self._format_messages(self.working[:-6])
        # ... 调用 AI 生成摘要 ...
        self.summaries.append("（摘要）")
        self.working = self.working[-6:]

    def _format_messages(self, messages: list) -> str:
        return "\n".join(f"{m['role']}: {m['content']}" for m in messages)
```

## 本章小结

- 三个层次：工作记忆、短期记忆、长期记忆
- 嵌入（Embedding）把文本变成向量
- ChromaDB 做本地向量存储
- RAG = 检索 + 生成，让 AI 基于你的知识回答
- 混合架构：多种记忆方式结合
