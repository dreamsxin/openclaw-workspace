## 3.1 Hello AI

最简单的 AI 调用：

```python
# code/ch03/hello_ai.py
from openai import OpenAI

client = OpenAI()  # 自动读取 OPENAI_API_KEY 环境变量

response = client.chat.completions.create(
    model="gpt-4",
    messages=[
        {"role": "user", "content": "用一句话介绍Python"}
    ]
)

print(response.choices[0].message.content)
```

运行：

```bash
python code/ch03/hello_ai.py
# 输出：Python 是一门简洁、优雅且功能强大的通用编程语言。
```

**就这么简单。5 行核心代码，完成一次 AI 交互。**

## 3.2 使用 Claude API

```python

import anthropic

    model="claude-sonnet-4-20250514",
    max_tokens=1024,

```

### Claude vs OpenAI SDK 对比

```python
# OpenAI 风格
response = client.chat.completions.create(
    model="gpt-4",
    messages=[...]
)
answer = response.choices[0].message.content

# Claude 风格
response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,    # Claude 必须指定
    messages=[...]
)
answer = response.content[0].text
```

## 3.4 多轮对话

AI 模型没有记忆——你必须把历史对话一起发送：

```python
# code/ch03/multi_turn.py
from openai import OpenAI

client = OpenAI()

# 对话历史
messages = [
    {"role": "system", "content": "你是一个友好的Python导师"}
]

def chat(user_input):
    messages.append({"role": "user", "content": user_input})

    response = client.chat.completions.create(
        model="gpt-4",
        messages=messages
    )

    answer = response.choices[0].message.content
    messages.append({"role": "assistant", "content": answer})
    return answer

# 对话
print(chat("什么是变量？"))
print(chat("能举个例子吗？"))
print(chat("那常量呢？"))  # 模型记得前面说了什么
```

### 对话历史结构

```json
[
    {"role": "system", "content": "你是一个友好的Python导师"},
    {"role": "user", "content": "什么是变量？"},
    {"role": "assistant", "content": "变量是用来存储数据的容器..."},
    {"role": "user", "content": "能举个例子吗？"},
    {"role": "assistant", "content": "当然！比如 name = \"小明\"..."},
    {"role": "user", "content": "那常量呢？"}
]
```

**每次请求都把完整历史发过去。** 这就是为什么对话越长，Token 消耗越大。

## 3.5 流式输出

长回复时，逐字输出体验更好：

```python
# code/ch03/streaming.py
from openai import OpenAI

client = OpenAI()

stream = client.chat.completions.create(
    model="gpt-4",
    messages=[
        {"role": "user", "content": "写一首关于Python的短诗"}
    ],
    stream=True  # 开启流式
)

# 逐块输出
for chunk in stream:
    content = chunk.choices[0].delta.content
    if content:
        print(content, end="", flush=True)

print()  # 换行
```

Claude 的流式写法：

```python
import anthropic

client = anthropic.Anthropic()

with client.messages.stream(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    messages=[{"role": "user", "content": "写一首关于Python的短诗"}]
) as stream:
    for text in stream.text_stream:
        print(text, end="", flush=True)
```

## 3.6 错误处理

真实世界里，API 调用会失败。必须处理：

```python
# code/ch03/error_handling.py
from openai import OpenAI, APIError, RateLimitError, APITimeoutError
import time

client = OpenAI()

def safe_call(messages, max_retries=3):
    """带重试的 AI 调用"""
    for attempt in range(max_retries):
        try:
            response = client.chat.completions.create(
                model="gpt-4",
                messages=messages,
                timeout=30  # 30秒超时
            )
            return response.choices[0].message.content

        except RateLimitError:
            wait = 2 ** attempt  # 指数退避
            print(f"速率限制，等待 {wait} 秒...")
            time.sleep(wait)

        except APITimeoutError:
            print(f"超时，重试 ({attempt + 1}/{max_retries})")

        except APIError as e:
            print(f"API 错误: {e}")
            break

    return "调用失败，请稍后重试"

# 使用
answer = safe_call([
    {"role": "user", "content": "你好"}
])
print(answer)
```

### 常见错误及处理

| 错误 | 原因 | 处理 |
|------|------|------|
| `RateLimitError` | 请求太频繁 | 等待后重试 |
| `APITimeoutError` | 网络超时 | 重试 |
| `AuthenticationError` | API 密钥无效 | 检查密钥 |
| `BadRequestError` | 请求格式错误 | 检查参数 |
| `ContextWindowExceeded` | 超出上下文长度 | 截断历史 |

## 3.7 一个完整的小工具

把本章学的内容综合起来——一个命令行聊天机器人：

```python
# code/ch03/chatbot.py
"""命令行聊天机器人 — 本章知识的综合运用"""

from openai import OpenAI
from rich.console import Console
from rich.markdown import Markdown

console = Console()
client = OpenAI()

messages = [
    {"role": "system", "content": "你是一个有帮助的助手，用简洁的中文回答。"}
]

def chat(user_input: str) -> str:
    messages.append({"role": "user", "content": user_input})

    stream = client.chat.completions.create(
        model="gpt-4",
        messages=messages,
        stream=True
    )

    collected = []
    for chunk in stream:
        content = chunk.choices[0].delta.content
        if content:
            collected.append(content)
            print(content, end="", flush=True)

    answer = "".join(collected)
    messages.append({"role": "assistant", "content": answer})
    print()
    return answer

def main():
    console.print("[bold green]AI 聊天机器人[/bold green]")
    console.print("输入 'quit' 退出\n")

    while True:
        try:
            user_input = input("你: ").strip()
            if user_input.lower() in ("quit", "exit", "q"):
                break
            if not user_input:
                continue

            console.print("\n[bold blue]AI:[/bold blue] ", end="")
            chat(user_input)
            print()

        except KeyboardInterrupt:
            break

    console.print("\n再见！👋")

if __name__ == "__main__":
    main()
```

参数的高阶函数，返回一个新的函数。
