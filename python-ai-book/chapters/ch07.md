# 第7章 构建第一个 AI Agent

> 从「对话」到「行动」——Agent 让 AI 不只是回答问题，还能做事。

## 7.1 什么是 Agent？

普通 AI 调用：你问 → AI 答 → 结束

Agent：你给目标 → AI 规划 → 调用工具 → 观察结果 → 继续规划 → 直到完成

```
用户: "帮我分析这个CSV文件"

普通AI: "请提供文件内容..."
Agent:  读取文件 → 查看结构 → 统计数据 → 生成报告 → 返回结果
```

### Agent 的核心循环

```
        ┌─────────────┐
        │   用户目标    │
        └──────┬──────┘
               ▼
        ┌─────────────┐
    ┌──→│    思考      │←──┐
    │   └──────┬──────┘   │
    │          ▼          │
    │   ┌─────────────┐   │
    │   │  选择工具    │   │
    │   └──────┬──────┘   │
    │          ▼          │
    │   ┌─────────────┐   │
    │   │  执行工具    │   │
    │   └──────┬──────┘   │
    │          ▼          │
    │   ┌─────────────┐   │
    │   │  观察结果    │   │
    │   └──────┬──────┘   │
    │          │          │
    │    完成？ │          │
    │    否 ───┘          │
    │                     │
    └─────────────────────┘
```

## 7.2 最简 Agent：ReAct 模式

ReAct = Reasoning + Acting（推理 + 行动）

```python
# code/ch07/react_agent.py
from openai import OpenAI
import json


class SimpleAgent:
    """简单的 ReAct Agent"""

    def __init__(self, model: str = "gpt-4"):
        self.client = OpenAI()
        self.model = model
        self.tools = {}
        self.system_prompt = """你是一个能使用工具的AI助手。

可用工具：
{tool_descriptions}

使用工具时，请输出JSON格式：
{{"tool": "工具名", "args": {{"参数": "值"}}}}

如果不需要工具，直接回答用户问题。
如果任务完成，在回复末尾加上 [DONE]。"""

    def register(self, name: str, func, description: str):
        """注册工具"""
        self.tools[name] = {
            "func": func,
            "description": description
        }

    def run(self, goal: str, max_steps: int = 10) -> str:
        """执行任务"""
        # 构建工具描述
        tool_desc = "\n".join(
            f"- {name}: {info['description']}"
            for name, info in self.tools.items()
        )

        messages = [
            {"role": "system", "content": self.system_prompt.format(
                tool_descriptions=tool_desc
            )},
            {"role": "user", "content": goal}
        ]

        for step in range(max_steps):
            # 思考
            response = self.client.chat.completions.create(
                model=self.model,
                messages=messages,
                temperature=0
            )

            reply = response.choices[0].message.content
            messages.append({"role": "assistant", "content": reply})

            # 检查是否完成
            if "[DONE]" in reply:
                return reply.replace("[DONE]", "").strip()

            # 尝试解析工具调用
            try:
                tool_call = self._extract_tool_call(reply)
                if tool_call:
                    # 执行工具
                    result = self._execute_tool(tool_call)
                    messages.append({
                        "role": "user",
                        "content": f"工具结果：{result}"
                    })
                    continue
            except Exception as e:
                messages.append({
                    "role": "user",
                    "content": f"工具执行错误：{e}"
                })
                continue

            # 没有工具调用也没有完成，继续
            break

        return reply

    def _extract_tool_call(self, text: str) -> dict | None:
        """从回复中提取工具调用"""
        # 寻找 JSON 块
        import re
        json_match = re.search(r'\{[\s\S]*\}', text)
        if json_match:
            data = json.loads(json_match.group())
            if "tool" in data:
                return data
        return None

    def _execute_tool(self, call: dict) -> str:
        """执行工具"""
        name = call["tool"]
        args = call.get("args", {})

        if name not in self.tools:
            return f"未知工具：{name}"

        return self.tools[name]["func"](**args)


# 使用示例
if __name__ == "__main__":
    agent = SimpleAgent()

    # 注册工具
    def calculate(expression: str) -> str:
        """安全计算数学表达式"""
        try:
            result = eval(expression, {"__builtins__": {}}, {})
            return str(result)
        except Exception as e:
            return f"计算错误：{e}"

    def get_time() -> str:
        """获取当前时间"""
        from datetime import datetime
        return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    agent.register("calculate", calculate, "计算数学表达式，如 '2+3*4'")
    agent.register("get_time", get_time, "获取当前时间")

    # 运行
    result = agent.run("现在几点了？3小时后是几点？")
    print(result)
```

## 7.3 Claude 的 Tool Use

Claude 有原生的工具调用支持，更优雅：

```python
# code/ch07/claude_tools.py
import anthropic
import json


client = anthropic.Anthropic()

# 定义工具
tools = [
    {
        "name": "get_weather",
        "description": "获取指定城市的天气",
        "input_schema": {
            "type": "object",
            "properties": {
                "city": {
                    "type": "string",
                    "description": "城市名称"
                }
            },
            "required": ["city"]
        }
    }
]

# 工具实现
def get_weather(city: str) -> str:
    # 模拟数据
    weather_data = {
        "北京": "晴，25°C",
        "上海": "多云，22°C",
    }
    return weather_data.get(city, f"{city}：未知")

# 对话
messages = [{"role": "user", "content": "北京今天天气怎么样？"}]

response = client.messages.create(
    model="claude-sonnet-4-20250514",
    max_tokens=1024,
    tools=tools,
    messages=messages
)

# 处理工具调用
for block in response.content:
    if block.type == "tool_use":
        result = get_weather(block.input["city"])
        print(f"工具调用：{block.name}({block.input}) → {result}")

        # 把结果告诉 Claude
        messages.append({"role": "assistant", "content": response.content})
        messages.append({
            "role": "user",
            "content": [{
                "type": "tool_result",
                "tool_use_id": block.id,
                "content": result
            }]
        })

        # Claude 生成最终回复
        final = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            tools=tools,
            messages=messages
        )
        print(final.content[0].text)
```

## 7.4 Agent 的设计原则

### 1. 工具要小而专

```python
# ❌ 不好：一个工具做太多事
def process_data(file_path, operation, column, filter_condition, ...):
    ...

# ✅ 好：每个工具做一件事
def read_csv(file_path): ...
def filter_rows(data, condition): ...
def calculate_stats(data, column): ...
def plot_chart(data, chart_type): ...
```

### 2. 提供清晰的工具描述

```python
# ❌ 不好
{"name": "query", "description": "查询"}

# ✅ 好
{
    "name": "query_database",
    "description": "执行SQL查询并返回结果。支持SELECT语句，返回JSON格式的行数据。"
}
```

### 3. 处理错误，不要崩溃

```python
def safe_tool(func):
    """工具装饰器：捕获异常并返回错误信息"""
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except Exception as e:
            return f"错误：{type(e).__name__}: {e}"
    return wrapper
```

## 本章小结

- Agent = AI + 工具 + 循环（思考→行动→观察）
- ReAct 是最基础的 Agent 模式
- Claude 有原生 Tool Use 支持
- 工具设计要小而专，描述要清晰
