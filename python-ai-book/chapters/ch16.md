# 第16章 上下文压缩

> 对话越长，Token 消耗越大，模型越「健忘」。压缩上下文是生产级应用的必修课。

## 16.1 为什么需要压缩？

一次真实的多轮对话：

```
轮次   Token 消耗（累计）
 1      ~200
 5      ~2,000
10      ~8,000
20      ~30,000
50      ~100,000+   ← 接近上下文窗口极限
```

**问题：**
- Token 费用线性增长
- 超出上下文窗口 = 丢失早期信息
- 长上下文推理质量下降（「大海捞针」问题）

**解决：压缩。用更少的 Token 保留更多信息。**

## 16.2 压缩策略总览

| 策略 | 原理 | 复杂度 | 信息保留率 |
|------|------|--------|-----------|
| **滑动窗口** | 只保留最近 N 轮 | ⭐ | 低 |
| **摘要压缩** | 用 AI 总结历史 | ⭐⭐ | 中高 |
| **Token 预算** | 按预算裁剪 | ⭐⭐ | 中 |
| **重要性筛选** | 保留关键消息 | ⭐⭐⭐ | 高 |
| **混合策略** | 组合使用 | ⭐⭐⭐ | 最高 |

## 16.3 滑动窗口

最简单的策略：只保留最近 N 轮对话。

```python
# code/ch16/sliding_window.py
from dataclasses import dataclass, field


@dataclass
class SlidingWindowCompressor:
    """滑动窗口压缩器"""

    max_turns: int = 10          # 保留最近 N 轮
    system_prompt: str = ""

    def compress(self, messages: list[dict]) -> list[dict]:
        """压缩消息列表"""
        result = []

        # 1. 保留 system prompt
        if self.system_prompt:
            result.append({"role": "system", "content": self.system_prompt})

        # 2. 提取非 system 消息
        conversation = [m for m in messages if m["role"] != "system"]

        # 3. 只保留最近 N 轮（每轮 = user + assistant）
        max_messages = self.max_turns * 2
        if len(conversation) > max_messages:
            dropped = len(conversation) - max_messages
            # 添加截断提示
            result.append({
                "role": "system",
                "content": f"（前 {dropped // 2} 轮对话已省略）"
            })
            conversation = conversation[-max_messages:]

        result.extend(conversation)
        return result


# 使用
compressor = SlidingWindowCompressor(max_turns=5)
messages = [
    {"role": "system", "content": "你是助手"},
    {"role": "user", "content": "第1轮"},
    {"role": "assistant", "content": "回复1"},
    # ... 更多轮次
    {"role": "user", "content": "第20轮"},
]
compressed = compressor.compress(messages)
```

**优点：** 简单、无额外 API 调用、确定性
**缺点：** 早期信息完全丢失

## 16.4 摘要压缩

用 AI 把早期对话压缩成摘要：

```python
# code/ch16/summarizer.py
import anthropic
from dataclasses import dataclass


@dataclass
class SummarizerCompressor:
    """摘要压缩器"""

    model: str = "claude-sonnet-4-20250514"
    max_recent_turns: int = 4      # 最近 N 轮保留原文
    summary_max_tokens: int = 300  # 摘要最大 Token

    def __post_init__(self):
        self.client = anthropic.Anthropic()

    def compress(self, messages: list[dict]) -> list[dict]:
        """压缩消息列表"""
        system = [m for m in messages if m["role"] == "system"]
        conversation = [m for m in messages if m["role"] != "system"]

        if len(conversation) <= self.max_recent_turns * 2:
            return messages  # 不需要压缩

        # 分割：需要摘要的部分 + 保留原文的部分
        split_point = len(conversation) - self.max_recent_turns * 2
        to_summarize = conversation[:split_point]
        to_keep = conversation[split_point:]

        # 生成摘要
        summary = self._generate_summary(to_summarize)

        # 重组
        result = system.copy()
        result.append({
            "role": "system",
            "content": f"以下是之前对话的摘要：\n\n{summary}"
        })
        result.extend(to_keep)

        return result

    def _generate_summary(self, messages: list[dict]) -> str:
        """用 AI 生成对话摘要"""
        conversation_text = "\n".join(
            f"{'用户' if m['role'] == 'user' else 'AI'}: {m['content']}"
            for m in messages
        )

        response = self.client.messages.create(
            model=self.model,
            max_tokens=self.summary_max_tokens,
            messages=[{
                "role": "user",
                "content": f"""请将以下对话压缩为简短摘要。

要求：
- 保留关键信息（用户的需求、决策、结论）
- 保留重要的技术细节
- 丢弃闲聊和重复内容
- 控制在 200 字以内

对话内容：
{conversation_text}"""
            }]
        )

        return response.content[0].text


# 使用
compressor = SummarizerCompressor(max_recent_turns=3)
compressed = compressor.compress(messages)
```

### 增量摘要（避免重复压缩）

```python
class IncrementalSummarizer:
    """增量摘要：只总结新内容，追加到已有摘要"""

    def __init__(self):
        self.client = anthropic.Anthropic()
        self.existing_summary = ""
        self.turn_count = 0
        self.compress_interval = 5  # 每 5 轮压缩一次

    def add_turn(self, user_msg: str, assistant_msg: str):
        self.turn_count += 1
        self.pending.append(("user", user_msg))
        self.pending.append(("assistant", assistant_msg))

        if self.turn_count % self.compress_interval == 0:
            self._compress_pending()

    def _compress_pending(self):
        """压缩待处理的消息，追加到已有摘要"""
        new_content = "\n".join(f"{r}: {c}" for r, c in self.pending)

        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=300,
            messages=[{
                "role": "user",
                "content": f"""已有摘要：
{self.existing_summary}

新对话内容：
{new_content}

请更新摘要，整合新内容，保持简洁。"""
            }]
        )

        self.existing_summary = response.content[0].text
        self.pending.clear()

    def get_context(self) -> str:
        context = self.existing_summary
        if self.pending:
            context += "\n\n最近对话：\n" + "\n".join(
                f"{r}: {c}" for r, c in self.pending
            )
        return context
```

## 16.5 Token 预算管理

给定一个 Token 预算，智能分配：

```python
# code/ch16/token_budget.py
import tiktoken
from dataclasses import dataclass


@dataclass
class TokenBudgetCompressor:
    """Token 预算压缩器"""

    total_budget: int = 8000      # 总预算
    system_reserve: int = 500     # system prompt 预留
    response_reserve: int = 1500  # 回复预留
    min_recent_turns: int = 3     # 至少保留最近 N 轮

    def __post_init__(self):
        self.encoder = tiktoken.encoding_for_model("gpt-4")

    def compress(self, messages: list[dict]) -> list[dict]:
        """按 Token 预算压缩"""
        available = self.total_budget - self.system_reserve - self.response_reserve

        system = [m for m in messages if m["role"] == "system"]
        conversation = [m for m in messages if m["role"] != "system"]

        # 从最新消息往前，直到填满预算
        selected = []
        used_tokens = 0

        for msg in reversed(conversation):
            msg_tokens = self._count_tokens(msg["content"])

            # 至少保留最近 N 轮
            if len(selected) >= self.min_recent_turns * 2:
                if used_tokens + msg_tokens > available:
                    break  # 超出预算，停止

            selected.insert(0, msg)
            used_tokens += msg_tokens

        # 如果有截断，添加提示
        if len(selected) < len(conversation):
            dropped = len(conversation) - len(selected)
            selected.insert(0, {
                "role": "system",
                "content": f"（{dropped} 条早期消息因 Token 限制被省略）"
            })

        return system + selected

    def _count_tokens(self, text: str) -> int:
        return len(self.encoder.encode(text))
```

## 16.6 重要性筛选

不是所有消息都同等重要。按重要性保留：

```python
# code/ch16/importance.py
from dataclasses import dataclass


@dataclass
class ImportanceCompressor:
    """基于重要性的压缩器"""

    max_messages: int = 20

    # 重要性规则
    HIGH_IMPORTANCE_KEYWORDS = [
        "错误", "bug", "修复", "决定", "结论", "重要",
        "记住", "注意", "警告", "TODO", "FIXME",
    ]

    def compress(self, messages: list[dict]) -> list[dict]:
        """按重要性筛选消息"""
        system = [m for m in messages if m["role"] != "system"]
        non_system = [m for m in messages if m["role"] != "system"]

        # 计算每条消息的重要性分数
        scored = []
        for i, msg in enumerate(non_system):
            score = self._calculate_score(msg, i, len(non_system))
            scored.append((score, i, msg))

        # 按分数排序，取 top N
        scored.sort(key=lambda x: x[0], reverse=True)
        selected = sorted(scored[:self.max_messages], key=lambda x: x[1])

        # 重组
        result = [m for m in messages if m["role"] == "system"]
        result.extend(msg for _, _, msg in selected)
        return result

    def _calculate_score(self, msg: dict, index: int, total: int) -> float:
        """计算消息重要性分数"""
        score = 0.0
        content = msg["content"].lower()

        # 1. 最近的消息更重要（时间衰减）
        recency = index / total  # 0-1，越大越新
        score += recency * 3

        # 2. 包含关键词的消息更重要
        for keyword in self.HIGH_IMPORTANCE_KEYWORDS:
            if keyword in content:
                score += 2

        # 3. 长消息通常包含更多信息
        score += min(len(content) / 500, 2)

        # 4. user 消息通常比 assistant 回复更重要
        if msg["role"] == "user":
            score += 1

        # 5. 包含代码的消息
        if "```" in content:
            score += 1.5

        return score
```

## 16.7 混合策略

生产环境中，通常组合多种策略：

```python
# code/ch16/hybrid.py
class HybridCompressor:
    """混合压缩策略"""

    def __init__(self, model: str = "claude-sonnet-4-20250514"):
        self.window = SlidingWindowCompressor(max_turns=20)
        self.summarizer = SummarizerCompressor(max_recent_turns=4)
        self.budget = TokenBudgetCompressor(total_budget=8000)
        self.importance = ImportanceCompressor(max_messages=30)

    def compress(self, messages: list[dict]) -> list[dict]:
        """多阶段压缩"""

        # 阶段1：滑动窗口（粗粒度裁剪）
        result = self.window.compress(messages)

        # 阶段2：如果仍然太长，用摘要压缩
        total_tokens = self._estimate_tokens(result)
        if total_tokens > 6000:
            result = self.summarizer.compress(result)

        # 阶段3：Token 预算兜底
        result = self.budget.compress(result)

        return result

    def _estimate_tokens(self, messages: list[dict]) -> int:
        return sum(len(m["content"]) // 2 for m in messages)  # 粗略估计
```

## 16.8 实战：带压缩的对话系统

```python
# code/ch16/compressed_chat.py
import anthropic
from hybrid import HybridCompressor


class CompressedChat:
    """带上下文压缩的对话系统"""

    def __init__(self, model: str = "claude-sonnet-4-20250514"):
        self.client = anthropic.Anthropic()
        self.model = model
        self.compressor = HybridCompressor(model)
        self.messages = []
        self.total_input_tokens = 0
        self.total_output_tokens = 0
        self.compression_count = 0

    def chat(self, user_input: str) -> str:
        # 添加用户消息
        self.messages.append({"role": "user", "content": user_input})

        # 压缩上下文
        original_count = len(self.messages)
        self.messages = self.compressor.compress(self.messages)
        if len(self.messages) < original_count:
            self.compression_count += 1
            print(f"  [压缩] {original_count} → {len(self.messages)} 条消息")

        # 调用 API
        response = self.client.messages.create(
            model=self.model,
            max_tokens=2000,
            messages=self.messages,
        )

        answer = response.content[0].text
        self.messages.append({"role": "assistant", "content": answer})

        # 统计
        self.total_input_tokens += response.usage.input_tokens
        self.total_output_tokens += response.usage.output_tokens

        return answer

    def stats(self) -> dict:
        return {
            "messages": len(self.messages),
            "input_tokens": self.total_input_tokens,
            "output_tokens": self.total_output_tokens,
            "compressions": self.compression_count,
        }
```

## 16.9 压缩效果对比

在 50 轮对话场景下的测试数据：

| 策略 | Token 消耗 | 信息保留率 | 延迟 |
|------|-----------|-----------|------|
| 不压缩 | ~150,000 | 100% | - |
| 滑动窗口(10轮) | ~30,000 | 20% | 0ms |
| 摘要压缩 | ~25,000 | 85% | ~2s |
| Token 预算 | ~20,000 | 70% | 0ms |
| 混合策略 | ~22,000 | 90% | ~2s |

## 本章小结

- 压缩是生产级 AI 应用的必备能力
- 滑动窗口最简单，摘要压缩效果最好
- Token 预算做兜底，重要性筛选保关键信息
- 混合策略：多阶段压缩，兼顾效果和性能
