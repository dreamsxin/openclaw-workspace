## 10.3 代码审查自动化

```python
# code/ch10/code_reviewer.py
import anthropic


class CodeReviewer:
    """AI 代码审查器"""

    def __init__(self):
        self.client = anthropic.Anthropic()

    def review(self, code: str, language: str = "Python") -> str:
        """审查代码"""
        prompt = f"""请审查以下 {language} 代码，按以下维度分析：

1. 🔴 **严重问题**（会导致崩溃或安全漏洞）
2. 🟡 **改进建议**（性能、可读性、最佳实践）
3. 🟢 **优点**（做得好的地方）

代码：
```{language}
{code}
```

请用 Markdown 格式输出审查结果。"""

        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=2000,
            messages=[{"role": "user", "content": prompt}]
        )
        return response.content[0].text

    def suggest_fix(self, code: str, issue: str) -> str:
        """针对特定问题建议修复"""
        prompt = f"""代码：
```python
{code}
```

问题：{issue}

请提供修复后的完整代码，只输出代码，不要解释。"""

        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1000,
            messages=[{"role": "user", "content": prompt}]
        )
        return response.content[0].text


if __name__ == "__main__":
    reviewer = CodeReviewer()

    sample_code = """
def get_user(user_id):
    query = f"SELECT * FROM users WHERE id = {user_id}"
    result = db.execute(query)
    return result
"""

    print(reviewer.review(sample_code))
```

## 10.5 项目级代码理解

```python
# code/ch10/project_analyzer.py
import anthropic
from pathlib import Path


class ProjectAnalyzer:
    """分析整个项目结构"""

    def __init__(self):
        self.client = anthropic.Anthropic()

    def analyze_project(self, project_path: str) -> str:
        """分析项目结构并生成说明"""
        # 收集项目文件
        files = self._collect_files(project_path)

        # 构建分析请求
        file_list = "\n".join(f"- {f}" for f in files.keys())

        prompt = f"""分析以下 Python 项目的结构：

项目文件：
{file_list}

请提供：
1. 项目概述（做什么用的）
2. 架构分析（模块关系）
3. 潜在问题
4. 改进建议

用中文回答。"""

        response = self.client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=2000,
            messages=[{"role": "user", "content": prompt}]
        )
        return response.content[0].text

    def _collect_files(self, path: str) -> dict[str, str]:
        """收集项目文件"""
        p = Path(path)
        files = {}
        for f in p.rglob("*.py"):
            if ".venv" not in str(f) and "__pycache__" not in str(f):
                try:
                    files[str(f.relative_to(p))] = f.read_text()[:2000]
                except Exception:
                    pass
        return files
```

## 10.6 工作流集成

把 Claude 集成到日常开发：

```python
# code/ch10/dev_workflow.py
"""开发工作流集成示例"""

import subprocess
import anthropic


client = anthropic.Anthropic()


def git_commit_with_ai():
    """用 AI 生成 git commit message"""
    # 获取 diff
    diff = subprocess.run(
        ["git", "diff", "--cached"],
        capture_output=True, text=True
    ).stdout

    if not diff.strip():
        print("没有暂存的更改")
        return

    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=100,
        messages=[{
            "role": "user",
            "content": f"为以下 git diff 生成简洁的 commit message（中文，不超过50字）：\n\n{diff}"
        }]
    )

    message = response.content[0].text
    print(f"建议的 commit message: {message}")

    confirm = input("使用这个消息？(y/n): ")
    if confirm.lower() == "y":
        subprocess.run(["git", "commit", "-m", message])


def explain_error(error_text: str):
    """用 AI 解释错误"""
    response = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=500,
        messages=[{
            "role": "user",
            "content": f"解释这个错误并给出修复建议：\n\n{error_text}"
        }]
    )
    print(response.content[0].text)
```
