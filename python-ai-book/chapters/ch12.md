# 第12章 自动化工作流

> 把多个 AI 步骤串联起来，构建端到端的自动化流程。

## 12.1 工作流的基本模式

### 线性流程

```
输入 → 步骤1 → 步骤2 → 步骤3 → 输出
```

### 分支流程

```
输入 → 判断 → 路径A → 输出
            → 路径B → 输出
```

### 循环流程

```
输入 → 步骤 → 检查 → 不满足 → 步骤
                    → 满足 → 输出
```

```python
# code/ch12/workflow.py
from dataclasses import dataclass, field
from typing import Callable, Any
from openai import OpenAI


@dataclass
class Step:
    name: str
    func: Callable
    description: str = ""


@dataclass
class Workflow:
    name: str
    steps: list[Step] = field(default_factory=list)
    context: dict = field(default_factory=dict)

    def add_step(self, name: str, func: Callable, desc: str = ""):
        self.steps.append(Step(name, func, desc))

    def run(self, initial_input: str) -> dict:
        """执行工作流"""
        self.context["input"] = initial_input
        current = initial_input

        for step in self.steps:
            print(f"▶ 执行步骤: {step.name}")
            try:
                result = step.func(current, self.context)
                self.context[step.name] = result
                current = result
                print(f"  ✅ 完成")
            except Exception as e:
                print(f"  ❌ 失败: {e}")
                self.context["error"] = str(e)
                break

        self.context["output"] = current
        return self.context


# 使用示例
client = OpenAI()


def analyze_topic(topic: str, ctx: dict) -> str:
    """步骤1：分析主题"""
    response = client.chat.completions.create(
        model="gpt-4",
        messages=[{
            "role": "user",
            "content": f"分析'{topic}'这个主题，列出3个关键要点"
        }]
    )
    return response.choices[0].message.content


def generate_content(keywords: str, ctx: dict) -> str:
    """步骤2：生成内容"""
    response = client.chat.completions.create(
        model="gpt-4",
        messages=[{
            "role": "user",
            "content": f"基于以下要点，写一篇500字的文章：\n{keywords}"
        }]
    )
    return response.choices[0].message.content


def review_content(content: str, ctx: dict) -> str:
    """步骤3：审查内容"""
    response = client.chat.completions.create(
        model="gpt-4",
        messages=[{
            "role": "user",
            "content": f"审查这篇文章，给出改进建议：\n{content}"
        }]
    )
    return response.choices[0].message.content


# 构建并运行
workflow = Workflow("文章生成")
workflow.add_step("分析", analyze_topic)
workflow.add_step("生成", generate_content)
workflow.add_step("审查", review_content)

result = workflow.run("Python 异步编程")
```

# code/ch12/conditional.py
@dataclass
class ConditionalStep:
    """条件分支步骤"""
    name: str
    condition: Callable[[Any], bool]
    true_step: Step
    false_step: Step

    def execute(self, input_data: str, context: dict) -> str:
        if self.condition(context):
            return self.true_step.func(input_data, context)
        else:
            return self.false_step.func(input_data, context)


# 示例：根据代码复杂度选择处理方式

## 12.4 循环与重试

```python
# code/ch12/retry.py
import time
from typing import Callable


def with_retry(
    func: Callable,
    max_attempts: int = 3,
    delay: float = 1.0,
    backoff: float = 2.0
) -> Callable:
    """重试装饰器"""
    def wrapper(*args, **kwargs):
        last_error = None
        for attempt in range(max_attempts):
            try:
                return func(*args, **kwargs)
            except Exception as e:
                last_error = e
                wait = delay * (backoff ** attempt)
                print(f"  重试 {attempt + 1}/{max_attempts}，等待 {wait}s")
                time.sleep(wait)
        raise last_error
    return wrapper


# 迭代改进循环
def iterative_improve(
    initial: str,
    evaluate: Callable[[str], float],
    improve: Callable[[str], str],
    target_score: float = 0.8,
    max_iterations: int = 5
) -> str:
    """迭代改进直到满足目标"""
    current = initial

    for i in range(max_iterations):
        score = evaluate(current)
        print(f"  迭代 {i + 1}: 分数 = {score:.2f}")

        if score >= target_score:
            print(f"  ✅ 达到目标分数 {target_score}")
            return current

        current = improve(current)

    print(f"  ⚠️ 达到最大迭代次数")
    return current
```

## 12.5 并行执行

```python
# code/ch12/parallel.py
import asyncio
from openai import AsyncOpenAI


async def parallel_tasks(tasks: list[dict]) -> list[str]:
    """并行执行多个 AI 任务"""
    client = AsyncOpenAI()

    async def run_task(task: dict) -> str:
        response = await client.chat.completions.create(
            model=task.get("model", "gpt-4"),
            messages=[{"role": "user", "content": task["prompt"]}]
        )
        return response.choices[0].message.content

    results = await asyncio.gather(*[run_task(t) for t in tasks])
    return results


# 使用示例
async def main():
    tasks = [
        {"prompt": "解释Python的GIL"},
        {"prompt": "解释Python的装饰器"},
        {"prompt": "解释Python的生成器"},
    ]

    results = await parallel_tasks(tasks)
    for i, result in enumerate(results):
        print(f"任务 {i + 1}: {result[:100]}...")


asyncio.run(main())
```
