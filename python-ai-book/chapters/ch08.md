# 第8章 工具调用

> 让 AI 能够「动手做事」——读文件、查数据库、调 API。

## 8.1 工具的本质

工具 = 一个 Python 函数 + 描述信息

```python
# 这就是一个工具
def search_files(query: str, directory: str = ".") -> list[str]:
    """在指定目录中搜索包含关键词的文件"""
    results = []
    for path in Path(directory).rglob("*"):
        if path.is_file() and query.lower() in path.read_text(errors="ignore").lower():
            results.append(str(path))
    return results
```

AI 需要知道的：
1. 工具叫什么名字（name）
2. 工具做什么（description）
3. 需要什么参数（parameters）
4. 怎么调用它

### 方式二：装饰器自动提取

```python
# code/ch08/tool_decorator.py
import inspect
from typing import get_type_hints


def tool(func):
    """装饰器：自动从函数签名生成工具定义"""
    hints = get_type_hints(func)
    sig = inspect.signature(func)

    properties = {}
    required = []

    for name, param in sig.parameters.items():
        prop = {"type": _python_type_to_json(hints.get(name, str))}
        if param.default is inspect.Parameter.empty:
            required.append(name)
        else:
            prop["default"] = param.default
        properties[name] = prop

    func._tool_def = {
        "name": func.__name__,
        "description": func.__doc__ or "",
        "input_schema": {
            "type": "object",
            "properties": properties,
            "required": required
        }
    }
    return func


def _python_type_to_json(t) -> str:
    mapping = {str: "string", int: "integer", float: "number", bool: "boolean"}
    return mapping.get(t, "string")

print(search_web._tool_def)
```

### 方式三：Pydantic 模型

```python
from pydantic import BaseModel, Field


class ReadFileInput(BaseModel):
    path: str = Field(description="文件路径")
    encoding: str = Field(default="utf-8", description="文件编码")


# 转换为工具定义
def schema_from_pydantic(model: type[BaseModel]) -> dict:
    schema = model.model_json_schema()
    return {
        "name": model.__name__.replace("Input", ""),
        "description": model.__doc__ or "",
        "input_schema": schema
    }
```

## 8.3 常用工具实现

### 文件操作工具

```python
# code/ch08/file_tools.py
from pathlib import Path


def read_file(path: str, encoding: str = "utf-8") -> str:
    """读取文件内容"""
    p = Path(path)
    if not p.exists():
        return f"错误：文件不存在 {path}"
    if p.stat().st_size > 100_000:
        return f"错误：文件太大 ({p.stat().st_size} bytes)"
    return p.read_text(encoding=encoding)


def write_file(path: str, content: str) -> str:
    """写入文件"""
    Path(path).parent.mkdir(parents=True, exist_ok=True)
    Path(path).write_text(content, encoding="utf-8")
    return f"已写入 {path} ({len(content)} 字符)"


def list_directory(path: str = ".", pattern: str = "*") -> list[str]:
    """列出目录内容"""
    p = Path(path)
    if not p.is_dir():
        return [f"错误：{path} 不是目录"]
    return [str(f.relative_to(p)) for f in p.glob(pattern)]
```

### 代码执行工具

```python
# code/ch08/code_tools.py
import subprocess
import tempfile


def run_python(code: str, timeout: int = 30) -> str:
    """在安全环境中执行 Python 代码"""
    with tempfile.NamedTemporaryFile(
        mode="w", suffix=".py", delete=False
    ) as f:
        f.write(code)
        f.flush()

        try:
            result = subprocess.run(
                ["python3", f.name],
                capture_output=True,
                text=True,
                timeout=timeout
            )
            output = result.stdout
            if result.stderr:
                output += f"\n[stderr]\n{result.stderr}"
            return output or "(无输出)"
        except subprocess.TimeoutExpired:
            return "错误：执行超时"
        except Exception as e:
            return f"错误：{e}"
```

## 8.4 工具注册表

集中管理所有工具：

```python
# code/ch08/tool_registry.py
class ToolRegistry:
    """工具注册表"""

    def __init__(self):
        self.tools = {}

    def register(self, name: str, func, definition: dict):
        self.tools[name] = {
            "func": func,
            "definition": definition
        }

    def get_definitions(self) -> list[dict]:
        """获取所有工具定义（给 AI 看）"""
        return [t["definition"] for t in self.tools.values()]

    def execute(self, name: str, args: dict) -> str:
        """执行工具"""
        if name not in self.tools:
            return f"未知工具：{name}"
        try:
            return str(self.tools[name]["func"](**args))
        except Exception as e:
            return f"工具执行错误：{e}"

    def list_tools(self) -> list[str]:
        return list(self.tools.keys())


# 使用
registry = ToolRegistry()
registry.register("read_file", read_file, {
    "name": "read_file",
    "description": "读取文件内容",
    "input_schema": {...}
})
```

## 8.5 安全考虑

### 不要给 AI 太多权限

```python
# ❌ 危险：直接执行任意代码
def run_code(code: str):
    exec(code)  # 永远不要这样！

# ✅ 安全：沙箱执行
def run_code_sandboxed(code: str):
    # 只允许安全的内置函数
    safe_builtins = {
        "print": print, "len": len, "range": range,
        "int": int, "float": float, "str": str,
        "list": list, "dict": dict, "set": set,
    }
    local_vars = {}
    exec(code, {"__builtins__": safe_builtins}, local_vars)
    return local_vars
```

### 工具白名单

```python
# 只注册安全的工具
SAFE_TOOLS = ["read_file", "search", "calculate"]

for tool_name in SAFE_TOOLS:
    if tool_name in all_tools:
        registry.register(tool_name, all_tools[tool_name], ...)
```
