## 14.2 数据加载与探索

```python
# code/ch14/data_analyst.py
"""自动化数据分析师"""

import pandas as pd
import anthropic
from pathlib import Path


class DataAnalyst:
    """AI 数据分析师"""

    def __init__(self, model: str = "claude-sonnet-4-20250514"):
        self.client = anthropic.Anthropic()
        self.model = model
        self.df = None
        self.history = []

    def load(self, file_path: str) -> str:
        """加载数据"""
        path = Path(file_path)
        suffix = path.suffix.lower()

        if suffix == ".csv":
            self.df = pd.read_csv(path)
        elif suffix in (".xlsx", ".xls"):
            self.df = pd.read_excel(path)
        elif suffix == ".json":
            self.df = pd.read_json(path)
        else:
            return f"不支持的格式：{suffix}"

        return f"已加载 {len(self.df)} 行 × {len(self.df.columns)} 列"

    def explore(self) -> str:
        """探索数据"""
        if self.df is None:
            return "请先加载数据"

        info = f"""## 数据概览

**形状**: {self.df.shape[0]} 行 × {self.df.shape[1]} 列

**列信息**:
| 列名 | 类型 | 非空数 | 示例值 |
|------|------|--------|--------|
"""
        for col in self.df.columns:
            dtype = self.df[col].dtype
            non_null = self.df[col].notna().sum()
            example = self.df[col].dropna().iloc[0] if non_null > 0 else "N/A"
            info += f"| {col} | {dtype} | {non_null} | {example} |\n"

        # 数值列统计
        numeric_cols = self.df.select_dtypes(include="number").columns
        if len(numeric_cols) > 0:
            info += "\n**数值统计**:\n"
            info += self.df[numeric_cols].describe().to_markdown()

        return info

    def ask(self, question: str) -> str:
        """用自然语言提问"""
        # 构建上下文
        context = self._build_context()

        prompt = f"""你是一个数据分析专家。根据以下数据信息回答问题。

{context}

问题：{question}

要求：
1. 如果需要代码，生成可执行的 Python 代码
2. 使用 pandas 和 matplotlib
3. 代码中包含中文注释
4. 如果需要可视化，生成图表代码
"""

        response = self.client.messages.create(
            model=self.model,
            max_tokens=2000,
            messages=[{"role": "user", "content": prompt}]
        )

        answer = response.content[0].text
        self.history.append({"question": question, "answer": answer})
        return answer

    def suggest_visualizations(self) -> str:
        """建议可视化方案"""
        context = self._build_context()

        prompt = f"""根据以下数据，建议3-5种合适的可视化方案：

{context}

对每种方案说明：
1. 图表类型
2. 适合展示什么
3. 具体的代码实现
"""

        response = self.client.messages.create(
            model=self.model,
            max_tokens=2000,
            messages=[{"role": "user", "content": prompt}]
        )
        return response.content[0].text

    def generate_report(self) -> str:
        """生成分析报告"""
        context = self._build_context()

        prompt = f"""根据以下数据，生成一份完整的数据分析报告：

{context}

报告结构：
1. 数据概览
2. 关键发现（3-5个）
3. 数据质量评估
4. 建议的下一步分析
"""

        response = self.client.messages.create(
            model=self.model,
            max_tokens=3000,
            messages=[{"role": "user", "content": prompt}]
        )
        return response.content[0].text

    def _build_context(self) -> str:
        """构建数据上下文"""
        if self.df is None:
            return "未加载数据"

        return f"""数据形状: {self.df.shape}
列: {list(self.df.columns)}
数据类型:
{self.df.dtypes.to_string()}

前5行:
{self.df.head().to_string()}

基本统计:
{self.df.describe().to_string()}
"""
```

## 14.3 代码生成与执行

```python
# code/ch14/code_executor.py
"""安全执行 AI 生成的代码"""

import pandas as pd
import matplotlib
matplotlib.use('Agg')  # 非交互模式
import matplotlib.pyplot as plt
import io
import contextlib


class CodeExecutor:
    """安全的代码执行器"""

    def __init__(self, df: pd.DataFrame):
        self.df = df
        self.sandbox = {
            "pd": pd,
            "df": df,
            "plt": plt,
            "print": print,
        }

    def execute(self, code: str) -> dict:
        """执行代码并捕获输出"""
        output = io.StringIO()
        result = {"success": False, "output": "", "error": "", "figure": None}

        try:
            with contextlib.redirect_stdout(output):
                exec(code, self.sandbox)
            result["success"] = True
            result["output"] = output.getvalue()

            # 检查是否有图表
            if plt.get_fignums():
                result["figure"] = plt.gcf()

        except Exception as e:
            result["error"] = f"{type(e).__name__}: {e}"

        return result


# 使用示例
analyst = DataAnalyst()
analyst.load("sales.csv")

# AI 生成代码
code = analyst.ask("按月份统计销售额并画折线图")

# 执行代码
executor = CodeExecutor(analyst.df)
result = executor.execute(code)

if result["figure"]:
    result["figure"].savefig("output.png")
```

### 自动异常检测

```python
def detect_anomalies(df: pd.DataFrame, column: str) -> pd.DataFrame:
    """检测异常值"""
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR

    anomalies = df[(df[column] < lower) | (df[column] > upper)]
    return anomalies
```

def auto_feature_engineering(df: pd.DataFrame) -> pd.DataFrame:
    """自动特征工程"""
    df = df.copy()

        df[f"{col}_log"] = np.log1p(df[col])
        df[f"{col}_zscore"] = (df[col] - df[col].mean()) / df[col].std()

    # 日期列：提取组件
    date_cols = df.select_dtypes(include="datetime").columns
    for col in date_cols:
        df[f"{col}_year"] = df[col].dt.year
        df[f"{col}_month"] = df[col].dt.month
        df[f"{col}_dayofweek"] = df[col].dt.dayofweek

    return df
