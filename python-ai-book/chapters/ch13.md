# 第13章 项目：智能代码助手

> 把前面学到的所有技能组合起来，构建一个完整的 AI 编程助手。

## 13.1 项目目标

构建一个命令行代码助手，能够：
- **读写本地文件** — 打开、编辑、保存代码文件
- **项目扫描** — 自动发现项目中的所有源码文件
- 分析代码问题
- 生成代码
- 重构代码（直接修改文件）
- 生成测试（自动保存到 test_xxx.py）
- 修复代码（读取文件 + 错误信息 → 写回修复结果）

### FileManager：本地文件读写

文件管理器是助手的基础层，负责所有磁盘操作：

```python
class FileManager:
    """本地文件读写管理"""

    def __init__(self, work_dir: str = "."):
        self.work_dir = Path(work_dir).resolve()
        self.work_dir.mkdir(parents=True, exist_ok=True)

    def read(self, file_path: str) -> str:
        """读取文件内容"""
        path = self._resolve(file_path)
        if not path.exists():
            raise FileNotFoundError(f"文件不存在: {path}")
        if path.stat().st_size > 1_000_000:
            raise ValueError(f"文件过大 ({path.stat().st_size} bytes)")
        return path.read_text(encoding="utf-8")

    def write(self, file_path: str, content: str, backup: bool = True) -> str:
        """写入文件，自动备份旧版本"""
        path = self._resolve(file_path)
        path.parent.mkdir(parents=True, exist_ok=True)
        if backup and path.exists():
            backup_path = path.with_suffix(f".bak.{datetime.now():%Y%m%d%H%M%S}")
            shutil.copy2(path, backup_path)
        path.write_text(content, encoding="utf-8")
        return str(path)

    def diff(self, file_path: str, new_content: str) -> str:
        """对比文件变更（unified diff）"""
```

### CodeAssistant：AI 核心引擎

助手类组合了 AI 能力和文件操作：

```python
class CodeAssistant:
    def __init__(self, work_dir: str = ".", model: str = "claude-sonnet-4-20250514"):
        self.client = anthropic.Anthropic()

    def analyze_file(self, file_path: str) -> str:
        content = self.files.read(file_path)
        return self.analyze(content, detect_language(file_path))

    def refactor_file(self, file_path: str, save: bool = False) -> str:
        content = self.files.read(file_path)
        result = self.refactor(content, detect_language(file_path))
        if save:
            self.save(file_path, result)
        return result

    def test_file(self, file_path: str, save: bool = False) -> str:
        content = self.files.read(file_path)
        result = self.generate_tests(content, detect_language(file_path))
        if save:
            test_path = Path(file_path).parent / f"test_{Path(file_path).stem}.py"
            self.save(str(test_path), result)
        return result

    def fix_file(self, file_path: str, error: str, save: bool = False) -> str:
        content = self.files.read(file_path)
        result = self.fix(content, error, detect_language(file_path))
        if save:
            self.save(file_path, result)
        return result

    def save(self, file_path: str, content: str) -> str:
        diff_text = self.files.diff(file_path, content)
        self.files.write(file_path, content)
        return f"已保存，变更:\n{diff_text}"
```

### ToolExecutor：工具执行层

文件操作之外，助手还需要执行 shell 命令、运行代码、调用 git 等。工具层把这些能力封装为独立模块，供 AI 调用：

```python
import subprocess
import shlex
from pathlib import Path


class ToolExecutor:
    """安全的工具执行器"""

    # 允许执行的命令白名单
    ALLOWED_COMMANDS = {
        "python", "python3", "pip", "pytest", "mypy", "ruff",
        "black", "isort", "git", "ls", "cat", "head", "tail",
        "grep", "find", "wc", "diff", "tree",
    }

    def __init__(self, work_dir: str = "."):
        self.work_dir = Path(work_dir).resolve()

    def run_command(self, command: str, timeout: int = 30) -> dict:
        """执行 shell 命令（白名单校验）"""
        parts = shlex.split(command)
        if not parts or parts[0] not in self.ALLOWED_COMMANDS:
            return {"success": False, "error": f"命令不允许: {parts[0] if parts else '(空)'}"}

        try:
            result = subprocess.run(
                parts,
                cwd=self.work_dir,
                capture_output=True,
                text=True,
                timeout=timeout,
            )
            return {
                "success": result.returncode == 0,
                "stdout": result.stdout,
                "stderr": result.stderr,
                "returncode": result.returncode,
            }
        except subprocess.TimeoutExpired:
            return {"success": False, "error": f"执行超时 ({timeout}s)"}
        except Exception as e:
            return {"success": False, "error": str(e)}

    def run_python(self, code: str, timeout: int = 30) -> dict:
        """执行 Python 代码片段"""
        with tempfile.NamedTemporaryFile(
            mode="w", suffix=".py", dir=self.work_dir, delete=False
        ) as f:
            f.write(code)
            f.flush()
            return self.run_command(f"python3 {f.name}", timeout)

    def run_tests(self, path: str = ".", timeout: int = 60) -> dict:
        """运行 pytest 测试"""
        return self.run_command(f"pytest {path} -v", timeout)

    def git_status(self) -> dict:
        """获取 git 状态"""
        return self.run_command("git status --short")

    def git_diff(self, file_path: str = "") -> dict:
        """获取 git diff"""
        cmd = f"git diff {file_path}" if file_path else "git diff"
        return self.run_command(cmd)

    def lint(self, file_path: str) -> dict:
        """代码检查（ruff）"""
        return self.run_command(f"ruff check {file_path}")

    def format_code(self, file_path: str) -> dict:
        """格式化代码（black）"""
        return self.run_command(f"black {file_path}")
```

工具层的安全设计：

| 原则 | 实现 |
|------|------|
| **白名单** | 只允许预定义的命令（python, git, pytest 等） |
| **超时** | 所有命令有超时限制，防止卡死 |
| **沙箱** | 在工作目录内执行，不访问系统其他位置 |
| **结果隔离** | 返回结构化 dict，不直接暴露进程 |

在 CodeAssistant 中集成工具层：

```python
class CodeAssistant:
    def __init__(self, work_dir: str = ".", model: str = "claude-sonnet-4-20250514"):
        self.client = anthropic.Anthropic()
        self.model = model
        self.files = FileManager(work_dir)     # 文件操作
        self.tools = ToolExecutor(work_dir)    # 工具执行
        self.history = []

    def run_tests(self, path: str = ".") -> str:
        """运行测试并用 AI 分析结果"""
        result = self.tools.run_tests(path)
        if result["success"]:
            return f"✅ 测试通过\n{result['stdout']}"
        else:
            # 失败时让 AI 分析原因
            analysis = self.explain_error(result["stderr"])
            return f"❌ 测试失败\n{result['stderr']}\n\n{analysis}"

    def lint_file(self, file_path: str) -> str:
        """检查代码并用 AI 解释问题"""
        result = self.tools.lint(file_path)
        if result["success"]:
            return f"✅ 无问题\n{result['stdout']}"
        return f"发现问题：\n{result['stdout']}"

    def auto_format(self, file_path: str) -> str:
        """自动格式化"""
        result = self.tools.format_code(file_path)
        return f"✅ 格式化完成" if result["success"] else f"❌ {result['error']}"
```
