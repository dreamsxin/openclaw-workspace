# 第6章 上下文管理

> AI 没有记忆——你得替它记住。这一章教你如何优雅地管理对话状态。

## 6.1 上下文窗口的限制

每个模型都有一个「上下文窗口」——单次能处理的最大 Token 数：

| 模型 | 上下文窗口 | 大约相当于 |
|------|-----------|-----------|
| GPT-4 | 128K Token | 一本小说 |
| Claude 3.5 | 200K Token | 一本书 |
| Qwen2.5-7B | 32K Token | 一篇论文 |
| Llama 3.1-8B | 128K Token | 一本小说 |

**对话超出窗口 = 模型「忘记」前面的内容。**

## 6.2 对话历史管理

### 基础方案：全量保存

```python
class ConversationManager:
    """简单的对话管理器"""

    def __init__(self, system_prompt: str = ""):
        self.messages = []
        if system_prompt:
            self.messages.append({
                "role": "system",
                "content": system_prompt
            })

    def add_user(self, content: str):
        self.messages.append({"role": "user", "content": content})

    def add_assistant(self, content: str):
        self.messages.append({"role": "assistant", "content": content})

    def get_messages(self) -> list:
        return self.messages.copy()

    def clear(self):
        system = self.messages[0] if self.messages[0]["role"] == "system" else None
        self.messages = [system] if system else []
```

### 滑动窗口：只保留最近 N 轮

```python
class SlidingWindowManager:
    """滑动窗口：只保留最近N轮对话"""

    def __init__(self, max_turns: int = 10, system_prompt: str = ""):
        self.max_turns = max_turns
        self.system_prompt = system_prompt
        self.history = []  # (user, assistant) 对

    def add_turn(self, user_msg: str, assistant_msg: str):
        self.history.append((user_msg, assistant_msg))
        # 只保留最近 N 轮
        if len(self.history) > self.max_turns:
            self.history = self.history[-self.max_turns:]

    def get_messages(self) -> list:
        messages = []
        if self.system_prompt:
            messages.append({"role": "system", "content": self.system_prompt})
        for user_msg, assistant_msg in self.history:
            messages.append({"role": "user", "content": user_msg})
            messages.append({"role": "assistant", "content": assistant_msg})
        return messages
```

## 6.3 Token 计数

知道用了多少 Token，才能控制成本和避免超出限制：

```python
# code/ch06/token_counter.py
import tiktoken


class TokenCounter:
    """Token 计数器"""

    def __init__(self, model: str = "gpt-4"):
        self.encoder = tiktoken.encoding_for_model(model)

    def count(self, text: str) -> int:
        return len(self.encoder.encode(text))

    def count_messages(self, messages: list) -> int:
        total = 0
        for msg in messages:
            # 每条消息有固定开销
            total += 4  # message overhead
            total += self.count(msg["content"])
        return total

    def truncate(self, text: str, max_tokens: int) -> str:
        """截断文本到指定 Token 数"""
        tokens = self.encoder.encode(text)
        if len(tokens) <= max_tokens:
            return text
        return self.encoder.decode(tokens[:max_tokens])


# 使用
counter = TokenCounter()
text = "这是一段测试文本"
print(f"Token 数: {counter.count(text)}")
```

## 6.4 智能摘要：压缩长对话

当对话太长时，用 AI 总结前面的内容：

```python
# code/ch06/summarizer.py
from openai import OpenAI


class SummarizingManager:
    """带摘要的对话管理器"""

    def __init__(self, client: OpenAI, model: str = "gpt-4"):
        self.client = client
        self.model = model
        self.messages = []
        self.summary = ""

    def add_turn(self, user_msg: str, assistant_msg: str):
        self.messages.append((user_msg, assistant_msg))

        # 超过5轮时自动摘要
        if len(self.messages) > 5:
            self._summarize()

    def _summarize(self):
        """用 AI 摘要前面的对话"""
        conversation = "\n".join(
            f"用户: {u}\nAI: {a}" for u, a in self.messages[:-2]
        )

        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": "将对话压缩为简短摘要，保留关键信息。"},
                {"role": "user", "content": conversation}
            ],
            max_tokens=200
        )

        self.summary = response.choices[0].message.content
        self.messages = self.messages[-2:]  # 只保留最近2轮

    def get_messages(self) -> list:
        messages = []
        if self.summary:
            messages.append({
                "role": "system",
                "content": f"之前的对话摘要：{self.summary}"
            })
        for user_msg, assistant_msg in self.messages:
            messages.append({"role": "user", "content": user_msg})
            messages.append({"role": "assistant", "content": assistant_msg})
        return messages
```

## 6.5 文件级记忆

把重要信息持久化到文件：

```python
# code/ch06/memory_manager.py
import json
from pathlib import Path
from datetime import datetime


class FileMemory:
    """文件级记忆管理"""

    def __init__(self, memory_dir: str = ".memory"):
        self.dir = Path(memory_dir)
        self.dir.mkdir(exist_ok=True)

    def save(self, key: str, value: str, category: str = "general"):
        """保存一条记忆"""
        path = self.dir / f"{category}.json"
        data = self._load(path)
        data[key] = {
            "value": value,
            "updated": datetime.now().isoformat()
        }
        path.write_text(json.dumps(data, ensure_ascii=False, indent=2))

    def recall(self, key: str, category: str = "general") -> str | None:
        """回忆一条记忆"""
        data = self._load(self.dir / f"{category}.json")
        entry = data.get(key)
        return entry["value"] if entry else None

    def recall_all(self, category: str = "general") -> dict:
        """回忆某个类别的所有记忆"""
        return {
            k: v["value"]
            for k, v in self._load(self.dir / f"{category}.json").items()
        }

    def _load(self, path: Path) -> dict:
        if path.exists():
            return json.loads(path.read_text())
        return {}


# 使用
memory = FileMemory()
memory.save("user_name", "Dreamszhu", "user")
memory.save("preferred_lang", "Python", "user")

print(memory.recall("user_name"))  # Dreamszhu
```

## 6.6 综合方案：完整的对话系统

把以上所有技术组合起来：

```python
# code/ch06/chat_system.py
from openai import OpenAI
from token_counter import TokenCounter
from summarizer import SummarizingManager
from memory_manager import FileMemory


class ChatSystem:
    """完整的对话系统：管理上下文、Token、记忆"""

    def __init__(self, model: str = "gpt-4", max_tokens: int = 4096):
        self.client = OpenAI()
        self.model = model
        self.max_tokens = max_tokens
        self.counter = TokenCounter(model)
        self.conversation = SummarizingManager(self.client, model)
        self.memory = FileMemory()

    def chat(self, user_input: str) -> str:
        # 1. 构建消息（包含记忆上下文）
        messages = self._build_messages(user_input)

        # 2. 检查 Token 预算
        input_tokens = self.counter.count_messages(messages)
        available = self.max_tokens - input_tokens - 500  # 留500给回复

        if available < 100:
            # Token 不够，需要压缩
            self.conversation._summarize()
            messages = self._build_messages(user_input)

        # 3. 调用 API
        response = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            max_tokens=min(available, 2000)
        )

        answer = response.choices[0].message.content

        # 4. 保存对话
        self.conversation.add_turn(user_input, answer)

        return answer

    def _build_messages(self, user_input: str) -> list:
        """构建完整的消息列表"""
        messages = self.conversation.get_messages()

        # 添加记忆上下文
        user_prefs = self.memory.recall_all("user")
        if user_prefs:
            context = "用户偏好：" + ", ".join(
                f"{k}={v}" for k, v in user_prefs.items()
            )
            messages.insert(1, {"role": "system", "content": context})

        messages.append({"role": "user", "content": user_input})
        return messages
```
